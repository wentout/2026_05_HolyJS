# POJO с прототипом

<div className='codesmall'>

```javascript
// POJO — обычный объект (Type)
const someObject = {
    name: 'myType',
    createdAt: new Date()
};

// Но мы можем добавить Interface!
Object.setPrototypeOf(someObject, {
    describe () {
        return `Type ${this.name} created at ${this.createdAt}`;
    }
});

console.log(someObject.describe());  // работает!
```

</div>

<Box sx={{marginTop: '5vh'}}>
<center>
<h3 style={{color: 'orange'}}>Type получил Interface через прототип!</h3>
</center>
</Box>

# Тот же результат через Constructor

<div className='codesmall'>

```javascript
// Constructor — это Interface
function Interface(name) {
    this.name = name;
    this.createdAt = new Date();
}

Interface.prototype.describe = function() {
    return `Interface ${this.name} created at ${this.createdAt}`;
};

const myInstance = new Interface('MyInterface');
console.log(myInstance.describe());  // тот же результат!
```
