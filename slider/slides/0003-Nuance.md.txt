# TS docs: &laquo;стилистический выбор&raquo;

<div className='codemedium'>

```typescript
// В TS оба варианта почти эквивалентны:
interface Point { x: number; y: number; }
type Point = { x: number; y: number; };

// Интерфейс может описывать функцию:
interface MyFn { (): void; }

// Type может описывать функцию:
type MyFn = () => void;
```

</div>

# Пример различия

<div className='codemedium'>

```typescript
    
// Interface содержит в себе и "до" и "после":
interface Factory {
        // " до " : с какой сущностью
        new (config: Config): Instance;
        // " после " : что можем делать
        create (): Product;
}
// Type — просто описание структуры сущности:
type Config = {
        name: string;
        count: number;
};
```

</div>
