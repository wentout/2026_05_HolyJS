# self reconstruct

<div className='codemedium' style={{marginTop: '-5vh'}}>

```javascript
const SelfReConstructor = function () {
    const Self = function () {
        if (new.target) {
            const Constructor = isItSelf ? Cstr : main;
            return new Constructor;
        }
        return Self;  // callable!
    };
    Object.setPrototypeOf(Self, root);
    Self.prototype.constructor = root;
    Self.value++;
    return Self;
};
```
