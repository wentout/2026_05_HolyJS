# await new — class + result

<div className='codemedium'>

```javascript
class MyPromise extends Promise {
    constructor(handler) { super(handler); }
}

const myItem = new MyPromise((resolve) => { resolve(123); });

// Мы можем await-ить результат new!
console.log(await myItem);  // 123

// И проверить наследование:
console.log(myItem instanceof Promise);  // true
```

</div>

<div className="slide-callout">
<h3><code className="inlineCode">new</code> + <code className="inlineCode">await</code> === новая семантика!</h3>
</div>

# можно и без наследования от Promise

<div className='codemedium'>

```javascript
class MyAsyncClass {
    field = undefined;
    constructor() {
        const self = this;
		return new Promise((resolve) => {
            setTimeout(() => {
                self.field = 123;
                resolve(self);
            }, 1000);
        });
    }
}
```

</div>

-----

# можно и без наследования от Promise

<div className='codemedium'>

```javascript
class MySubAsyncClass extends MyAsyncClass {
    constructor() {
        // да, super возвращает результат ! 
        const promise = super();
        return new Promise(async (resolve) => {
            // тут item содержит  this !!!
            const item = await promise;
            console.log(item.field); // 123
            resolve(item);
        });
    }
}
```

</div>


-----

# запустим и посмотрим

<div className='codemedium'>

```javascript

    ( async () => {

        // создадим ожидание наследника
        const successor = new MySubAsyncClass;

        // создадим сам экземпляр наследника
        const item = await successor;

        // проверим, что всё получилось 
        console.log(item.field); // 123

} )();

```
